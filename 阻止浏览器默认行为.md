## 只能输入数字案例

```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>只能输入数字</title>
</head>
<body>
   只能输入数字:<input type="text">  
    <script>
	var input = document.getElementsByTagName("input")[0];
	
	input.onkeydown = function (e) {
		e = window.event || e;
		/*debugger;
		console.log(e.key);
		if (isNaN(e.key) && e.keyCode != "Backspace") {
			stopDefault(e);
		} */
		console.log(e.keyCode);
		if ((e.keyCode < 48 || e.keyCode > 57) && e.keyCode != 8) {
			stopDefault(e);
		}

	}
	
	
	
	//阻止浏览器默认行为的兼容性写法
	function stopDefault(e){
		//这个是防止如果上面的 stopDefault(e)没有传参数e,则if (!"") 里的判断为true，
		//就不执行下面的代码，就不会报错了，如果把空的字符串放在下面的条件语句中，会报错
		if (!e) return;  

		if (e.preventDefault) {
			e.preventDefault();
		} else {
			e.returnValue = false; 
		}
	}
    </script> 
</body>
</html>
```



## 阻止a标签默认行为

```
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>阻止a标签的默认行为</title>
</head>
<body>
  阻止a标签:<a href="http://www.baidu.com">点击</a>
  <input type="text">

<script>

    var a = document.getElementsByTagName("a")[0];
    var input = document.getElementsByTagName("input")[0];

    a.onclick = function (e) {
        e = window.event || e;
        stopDefault(e);

        if (input.value == "11") {
            location.href = "http://www.taobao.com";
        }
    }


    //阻止浏览器默认行为的兼容性写法
    function stopDefault(e){
        //debugger;
        if (!e) return;
        if (e.preventDefault) {
            e.preventDefault();
        } else {
            e.returnValue = false;
        }
    }
</script>
</body>
</html>
```
